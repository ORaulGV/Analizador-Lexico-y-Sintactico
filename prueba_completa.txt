class Calculadora {
    // AccessOpt (public/private), Type (int/bool), ID y ArrOpt (arrays)
    public int resultado;
    private bool encendido;
    int[] historial; 

    // StaticOpt, RetType (void), ParamListOpt (vacio) y Block
    public static void main() {
        // Declaracion simple y con inicializacion (VarDeclTail)
        int a;
        int b = 10;
        bool check = true;

        // Jerarquia aritmetica (multiplicacion antes que suma)
        a = 5 + 5 * 2; 

        // Operadores relacionales (>) y bloques anidados
        if (a > b) {
            a = a - 1;
        } else {
            a = a + 1;
        }

        // Operadores de igualdad (==)
        while (check == true) {
            // Acceso a arreglos ([]) y asignacion
            historial[0] = a;
            check = false;
        }
        return;
    }

    // ParamList (multiples argumentos) y retorno de expresion
    int sumar(int x, int y) {
        return x + y;
    }
}
